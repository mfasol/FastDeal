create VIEW V_SALES_CONTRIBUTION AS SELECT
  TRANSACTION_DATE ,
  EXTERNAL_TRANSACTION_ID ,
  PRODUCT_KEY ,
  QUANTITY ,
  TRANSACTION_PRICE ,
  TRANSACTION_ADDITIONAL_COS,
  TRANSACTION_CHANNEL,
  TRANSACTION_CHANNEL_COUNTRY,
  CURRENCY,
  TRANSACTION_ID,
  TRANSACTION_LINE_ID,
  TRANSACTION_LINE_UUID,
  TRANSACTION_STATUS,
  ITEM_ID,
  TRANSACTION_TIMESTAMP,
  TRANSACTION_LOCK,
  ITEM_UUID
 FROM SALES_LEDGER WHERE TRANSACTION_STATUS = 'SALE';
