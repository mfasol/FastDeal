create OR REPLACE VIEW V_SALES_CONTRIBUTION AS (SELECT
  SALES_LEDGER.ITEM_ID, SALES_LEDGER.PRODUCT_KEY, SALES_LEDGER.ITEM_UUID,
  SALES_LEDGER.TRANSACTION_PRICE AS SALES, ROUND(INVENTORY_ITEMS_GB_AMAZON.ITEM_COS,4) AS COS,
  ROUND((SALES_LEDGER.TRANSACTION_PRICE -  INVENTORY_ITEMS_GB_AMAZON.ITEM_COS),4) AS PROFIT
FROM SALES_LEDGER LEFT JOIN INVENTORY_ITEMS_GB_AMAZON ON
 SALES_LEDGER.ITEM_ID = INVENTORY_ITEMS_GB_AMAZON.CONCATENATED_PRIMARY_KEY AND
 SALES_LEDGER.ITEM_UUID = INVENTORY_ITEMS_GB_AMAZON.SINGLE_ITEM_UUID
WHERE SALES_LEDGER.TRANSACTION_STATUS='SALE');
